# Nested

You can nest panels inside of each other to create complex layouts.

<div className="my-6">
  <NestedExample />
</div>

```tsx
import { PanelGroup, Panel, PanelResizer } from "@window-splitter/react";

function NestedExample() {
  return (
    <PanelGroup
      orientation="horizontal"
      style={{
        border: "1px solid rgba(0, 0, 0, 0.3)",
        borderRadius: 12,
        height: 400,
      }}
    >
      <Panel id="panel-1" min="10%">left</Panel>
      <PanelResizer id="resizer-1" />
      <Panel id="panel-2" min="10%">
        <PanelGroup orientation="vertical">
          <Panel id="panel-2-1" min="10%">top</Panel>
          <PanelResizer id="resizer-2-1" />
          <Panel id="panel-2-2" min="10%">
            <PanelGroup orientation="horizontal">
              <Panel id="panel-2-2-1" min="20%">left</Panel>
              <PanelResizer id="resizer-2-2-1" />
              <Panel id="panel-2-2-2" min="20%">right</Panel>
            </PanelGroup>
          </Panel>
        </PanelGroup>
      </Panel>
      <PanelResizer id="resizer-2" />
      <Panel id="panel-3" min="10%">right</Panel>
    </PanelGroup>
  );
}
```
